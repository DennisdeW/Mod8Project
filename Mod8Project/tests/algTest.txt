program simpleAlgorithmsTest;

def int main() {
	int result = 0;
	if (!testGcd()) {
		result = 1;
	}
	if (!testFac()) {
		result = 2;
	}
	out(result);
	return result;
}

def bool testGcd() {
	if (gcd(6, 9) != 3) {
		return false;
	}
	if (gcd(100, 20) != 20) {
		return false;
	}
	if (gcd(17, 5) != 1) {
		return false;
	}
	return true;
}

def int gcd(int x, int y) {
	int t = 0;
	while (y != 0) {
		t = y;
		y = x % y;
		x = t;
	}
	return x;
}

def bool testFac() {
	if (fac(3) != 6) {
		return false;
	}
	if (fac(4) != 24) {
		return false;
	}
	if (fac(5) != 120) {
		return false;
	}
	if (fac(10) != 3628800) {
		return false;
	}
	if (fac(0) != 1) {
		return false;
	}
	return true;
}

def int fac(int n) {
	int res = 1;
	for (int i = 1; i <= n; i = i + 1) {
		res = res * i;
	}
	return res;
}
